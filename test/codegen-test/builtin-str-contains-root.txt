#include <stdbool.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct str {
  char *d;
  size_t l;
};

void *alloc (size_t);
struct str str_alloc (const char *);
bool str_contains (struct str, struct str);

void *alloc (size_t l) {
  void *r = malloc(l);
  if (r == NULL) {
    fprintf(stderr, "Error: failed to allocate %zu bytes" THE_EOL, l);
    exit(EXIT_FAILURE);
  }
  return r;
}
struct str str_alloc (const char *r) {
  size_t l = strlen(r);
  char *d = alloc(l);
  memcpy(d, r, l);
  return (struct str) {d, l};
}
bool str_contains (struct str self, struct str n1) {
  bool r = n1.l == 0;
  if (!r) {
    for (size_t i = 0; i <= self.l; i++) {
      if (memcmp(&self.d[i], n1.d, n1.l) == 0) {
        r = true;
        break;
      }
    }
  }
  free(self.d);
  free(n1.d);
  return r;
}

int main () {
  str_contains(str_alloc(""), str_alloc(""));
  str_contains(str_alloc("t"), str_alloc(""));
  str_contains(str_alloc("test on test"), str_alloc(""));
  str_contains(str_alloc(""), str_alloc("t"));
  str_contains(str_alloc("t"), str_alloc("t"));
