void *alloc (size_t l) {
  void *r = malloc(l);
  if (r == NULL) {
    fprintf(stderr, "Error: failed to allocate %zu bytes" THE_EOL, l);
    exit(EXIT_FAILURE);
  }
  return r;
}
struct str str_alloc (const char *r) {
  size_t l = strlen(r);
  char *d = alloc(l);
  memcpy(d, r, l);
  return (struct str) {d, l};
}
char *str_at (struct str s, int64_t i) {
  if ((i >= 0 && i >= s.l) || (i < 0 && i < -((int64_t) s.l))) {
    fprintf(stderr, "Error: index %" PRId64 " out of string bounds" THE_EOL, i);
    exit(EXIT_FAILURE);
  }
  return i < 0 ? &s.d[s.l + i] : &s.d[i];
}
void str_free (struct str s) {
  free(s.d);
}
