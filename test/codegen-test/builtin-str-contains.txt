struct str str_alloc (const char *r) {
  size_t l = strlen(r);
  char *d = alloc(l);
  memcpy(d, r, l);
  return (struct str) {d, l};
}
void str_free (struct str s) {
  free(s.d);
}
bool str_contains (struct str self, struct str n1) {
  bool r = n1.l == 0;
  if (!r) {
    for (size_t i = 0; i <= self.l; i++) {
      if (memcmp(&self.d[i], n1.d, n1.l) == 0) {
        r = true;
        break;
      }
    }
  }
  free(self.d);
  free(n1.d);
  return r;
}
