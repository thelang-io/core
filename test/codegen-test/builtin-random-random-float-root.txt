double random_randomFloat (unsigned char o1, double n1, unsigned char o2, double n2) {
  double a = o1 == 0 ? 0 : n1;
  double b = o2 == 0 ? 1 : n2;
  unsigned char d[4];
  if (RAND_bytes(d, 4) != 1) {
    fprintf(stderr, "Error: failed to generate random bytes" THE_EOL);
    exit(EXIT_FAILURE);
  }
  uint32_t r = (d[0] << 24) | (d[1] << 16) | (d[2] << 8) | d[3];
  return a + r / ((double) UINT32_MAX / (b - a));
}

int main () {
  random_randomFloat(0, 0, 0, 0);
  random_randomFloat(1, -100, 1, 100);
}
