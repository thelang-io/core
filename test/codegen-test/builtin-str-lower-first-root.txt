#include <ctype.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

struct str {
  char *d;
  size_t l;
};

void *alloc (size_t);
struct str str_alloc (const char *);
void str_free (struct str);
struct str str_lower_first (struct str);

void *alloc (size_t l) {
  void *r = malloc(l);
  if (r == NULL) {
    fprintf(stderr, "Error: failed to allocate %zu bytes" THE_EOL, l);
    exit(EXIT_FAILURE);
  }
  return r;
}
struct str str_alloc (const char *r) {
  size_t l = strlen(r);
  char *d = alloc(l);
  memcpy(d, r, l);
  return (struct str) {d, l};
}
void str_free (struct str s) {
  free(s.d);
}
struct str str_lower_first (struct str s) {
  if (s.l != 0) s.d[0] = tolower(s.d[0]);
  return s;
}

int main () {
  str_free((struct str) str_lower_first(str_alloc("")));
  str_free((struct str) str_lower_first(str_alloc("a")));
  str_free((struct str) str_lower_first(str_alloc("test")));
  str_free((struct str) str_lower_first(str_alloc("A")));
  str_free((struct str) str_lower_first(str_alloc("TEST_TEST09$/!")));
}
