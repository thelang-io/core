void *alloc (size_t l) {
  void *r = malloc(l);
  if (r == NULL) {
    fprintf(stderr, "Error: failed to allocate %zu bytes" THE_EOL, l);
    exit(EXIT_FAILURE);
  }
  return r;
}
struct str str_alloc (const char *r) {
  size_t l = strlen(r);
  char *d = alloc(l);
  memcpy(d, r, l);
  return (struct str) {d, l};
}
char *str_cstr (const struct str s) {
  char *d = alloc(s.l + 1);
  memcpy(d, s.d, s.l);
  d[s.l] = '\0';
  return d;
}
uint64_t str_toU64 (struct str self, unsigned char o1, int32_t n1) {
  if (o1 == 1 && (n1 < 2 || n1 > 36) && n1 != 0) {
    fprintf(stderr, "Error: radix %" PRId32 " is invalid, must be >= 2 and <= 36, or 0" THE_EOL, n1);
    exit(EXIT_FAILURE);
  }
  char *c = str_cstr(self);
  char *e = NULL;
  errno = 0;
  unsigned long long r = strtoull(c, &e, o1 == 0 ? 10 : n1);
  if (errno == ERANGE || UINT64_MAX < r) {
    fprintf(stderr, "Error: value `%s` out of range" THE_EOL, c);
    exit(EXIT_FAILURE);
  } else if (errno != 0 || e == c || *e != 0 || self.d[0] == '-') {
    fprintf(stderr, "Error: value `%s` has invalid syntax" THE_EOL, c);
    exit(EXIT_FAILURE);
  }
  free(c);
  free(self.d);
  return (uint64_t) r;
}
